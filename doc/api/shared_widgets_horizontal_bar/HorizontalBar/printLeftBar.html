  return Positioned(
    height: 10,
    left: 10,
    child: Container(
      width: screenWidth * 0.45,
      height: 10,
      decoration: BoxDecoration(
        color: ColorConstants.appColor,
        borderRadius: BorderRadius.circular(10),
      ),
    ),
  );
